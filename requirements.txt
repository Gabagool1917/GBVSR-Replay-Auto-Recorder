# GBVSR Auto Recorder - runtime dependencies
# Install with:  pip install -r requirements.txt
# Targets Windows + Python 3.10+ (pydirectinput and the automation features
# only work on Windows; combining recordings works on any OS).

PySide6>=6.5              # GUI toolkit
pyautogui>=0.9.54         # screen template matching (menu/loading detection) + cursor-free pixel checks
pydirectinput>=1.0.4      # Windows-only: sends real DirectInput key events GBVSR's game engine will see
keyboard>=0.13.5          # global hotkey listening (hold 's' to pause, 'e' to stop)
moviepy>=2.0,<3           # trims and crossfades the per-game recordings into one video
opencv-python>=4.8        # required by pyautogui's confidence= fuzzy-matching (template images aren't pixel-perfect)
